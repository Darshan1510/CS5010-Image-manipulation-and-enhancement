load D:\test_images\paris-test.png paris

#brighten paris by adding 10  
brighten 10 paris paris-brighter 

#flip paris vertically
vertical-flip paris paris-vertical

#luma paris vertical

#save paris-brighter
save D:\test_images\paris-brighter.png paris-brighter

save D:\test_images\paris-gs.jpg paris-greyscale
